Your AI agent must define a get_move() function.
It receives a chess.Board object and must return
a valid UCI move string (e.g. 'e2e4', 'g1f3').
"""
import chess
import random

def get_move(board: chess.Board) -> str:
    """Return a UCI move string for the given position."""
    legal_moves = list(board.legal_moves)
    # Example: pick a random legal move
    move = random.choice(legal_moves)
    return move.uci()
